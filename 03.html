1. 父子组件的通信

使用prop从父组件向子组件传递数据，注意：camelCased vs kebab-case和单向数据流

使用自定义事件把数据从子组件传递给父组件，注意：用于父组件监听的自定义事件名称必须全部小写


Vue.component('child', {
  // camelCase in JavaScript
  props: ['myMessage'],
  template: '{{ myMessage }}',
  methods: {
    increment: function() {
      // do something ...
      this.$emit('incrementtotal')
    }
  }
})

<child my-message="hello!" v-on:incrementtotal="incrementTotal">

new Vue({
  el: '#example',
  data: {
    total: 0
  },
  methods: {
    incrementTotal: function () {
      this.total += 1
    }
  }
})
2. 非父子组件通信：全局事件管理bus
      
// bus.js
const bus = new Vue()
export default bus

// 组件A
import bus from 'bus.js'
new Vue({
  methods: {
    action: function() {
      bus.$emit('eventName',value)
    }
  }
})

// 组件B
import bus from 'bus.js' 
new Vue({
  created: function() {
    bus.$on('eventName',function(value) {
      // do something ...
    })
  }
})
